# NaCl example
lattice_vectors = np.array([
    [0.0, 0.5, 0.5],
    [0.5, 0.0, 0.5],
    [0.5, 0.5, 0.0],
])
atoms = {
    "Na": np.array([[0.0, 0.0, 0.0]]),
    "Cl": np.array([[0.5, 0.5, 0.5]]),
}
supa = np.identity(3, dtype=int)
orbitals = "p"
structure = CrystalFTG(vec=lattice_vectors, atoms=atoms, supa=supa, orbitals=orbitals)
