Lone Irreducible Derivatives (LID) Approach

Here we demonstrate computing phonon of rocksalt with a FTG of 2\(\hat{1}\) using LID approach.

import numpy as np

from principia_materia.io_interface.qpoints_io import tuple_to_str

from principia_materia.phonon_id.lid_mesh import LIDMesh
from principia_materia.phonon_id.fourier_interpolation import (
    save_FourierInterpolation_to_hdf5,
    FourierInterpolation,
    )

    get_structure,
    CrystalFTG,
    save_structure,
    )

[2]:

POSCAR = "./nacl_lid/POSCAR"
EPSILON = "./nacl_lid/epsilon.yml"

ORDER = 2
PG = "Oh"
MESH = np.diag([2, 2, 2])
DELTA = np.arange(0.02, 0.21, 0.02)

COMPUTE_ENGINE = "vasp"

ROOT_DIR = "."
