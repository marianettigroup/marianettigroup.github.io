def get_wigner_seitz_cell(self, center):
  ws_cell_translation = []
  ws_cell_atoms = []
  ws_cell_weights = []
  nrange = 0
  while not np.allclose(np.sum(ws_cell_weights), self.natoms):
    tvecs = np.array(list(\
              set(itertools.product(np.arange(- nrange, nrange + 1), repeat=self.dim)) \
              - set(itertools.product(np.arange(- (nrange - 1), nrange), repeat=self.dim))
             )
          )
    for i in range(self.natoms):
      weights = self.ws_weight(tvecs + self.positions[i] - center)
      for j in weights.nonzero()[0]:
        ws_cell_translation.append(np.dot(tvecs[j], self.supa) + self.supa_translations[i])
        ws_cell_atoms.append(i)
        ws_cell_weights.append(weights[j])
    nrange += 1
    if nrange > 5:
      raise RuntimeError("Failed to find all atoms in the Wigner-Seitz cell.")

  atoms_weight = np.zeros(self.natoms)
  for i, j in zip(ws_cell_atoms, ws_cell_weights):
    atoms_weight[i] += j
  if not np.allclose(atoms_weight, np.ones(self.natoms)):
    raise ValueError("Error in determining WS cell, not all atoms have total weight of 1.")

  ws_cell_translation =  np.array(ws_cell_translation)
  ws_cell_atoms =  np.array(ws_cell_atoms)
  ws_cell_weights = np.array(ws_cell_weights)
  return ws_cell_translation, ws_cell_atoms, ws_cell_weights
